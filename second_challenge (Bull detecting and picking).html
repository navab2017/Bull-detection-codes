def color_blob():
    hsv = cv2.cvtColor(p3dx.image, cv2.COLOR_RGB2HSV)
    mask = cv2.inRange(hsv, lower, upper)
    mask[:, 0:60] = 0 # mask out left outliers
    mask[:, 90:150] = 0 # mask out right outliers
    
    M = cv2.moments(mask)
    area = M['m00']


    if area > 0:
        cx = int(M['m10']/area)
        cy = int(M['m01']/area)
    else:
        cx = None
        cy = None
    return area, cx, cy


def is_blob_centered():
    area, cx, cy = color_blob()
    if area > 0 and cx >= 70 and cx < 80:
        return True
    else:
        return False


def is_ball_close():
    area, cx, cy = color_blob()
    if area > 0 and cy >= 90:
        return True
    else:
        return False

def is_target_close():
    area, cx, cy = color_blob()
    if area > 0 and cy >= 70:
        return True
    else:
        return False
    
def is_target_reached():
    area, cx, cy = color_blob()
    if area > 0 and cy >= 50:
        return True
    else:
        return False


def find_and_get_ball():
    # find the ball
    p3dx.tilt(-0.35)
    p3dx.sleep(1)
    while not is_blob_centered():
        p3dx.move(-0.5,0.5)
    p3dx.stop()
    print('Ball detected')
    
    #open gripper and move towards ball
    p3dx.gripper(0.05,0.1)
    p3dx.sleep(2)
    while not is_ball_close():
        if not is_blob_centered(): # if blob goes out of sight
            # rotate slightly to thr right before continuing 
            # to look for it (prevents unnecessary changing ball selection)
            p3dx.move(0.5,-0.5) 
            p3dx.sleep(0.5)
            
        while not is_blob_centered():
            p3dx.move(-0.5,0.5)
        p3dx.move(3.0,3.0)
    p3dx.stop()
    print('Ball is close')
    
    #Tilt down to see ball
    p3dx.tilt(-0.47)
    #center if required
    while not is_blob_centered():
        p3dx.move(-0.5,0.5)
    p3dx.stop()
    print('Fine tune Ball detection')
    
    #move in closer
    count = 0
    while not is_ball_close() and (count<30):
        count = count +1
        #print "count = ", count
        p3dx.move(0.5,0.5)
    p3dx.stop()
    if (count >29):
        print('Ball is lost, backup and retry')
        ball_secured = False
        p3dx.move(-1.0,-1.0)
        p3dx.sleep(3)
        p3dx.stop()
    else:
        print('Ball is very close')
        #move in and pick up ball and stop
        p3dx.move(1.0,1.0)
        p3dx.gripper(0.05,0.0)
        p3dx.stop()
        p3dx.sleep(2)
        print('Ball is acquired')
    
        # raise gripper
        p3dx.gripper(-0.00,0.0)
    
        #check ball is held
        p3dx.tilt(-0.47) # look down
        p3dx.move(-1.0,-1.0)# and back up a little
        p3dx.sleep(2)
        area, cx, cy = color_blob()
        #print "area, cx, cy", area, cx, cy
        if (area>10000):
            ball_secured = True
            print "Ball secured"
        else:
            ball_secured = False
            print "Ball fumbled"
      
    return ball_secured

def find_target_and_put_ball():
    #
    
    #look up
    p3dx.tilt(0.0)
    p3dx.sleep(1)
    
    #center on target
    while not is_blob_centered():
        p3dx.move(-0.5,0.5)
    p3dx.stop()
    print('Target detected')
    
    #move towards target
    while not is_target_close():
        p3dx.move(2.0,2.0)
    p3dx.stop()
    print('Target is close')
    
    #look down
    p3dx.tilt(-0.47)
    p3dx.sleep(1)
    #and move in close
    while not is_target_reached():
        p3dx.move(0.5,0.5)
    
    print('Target is reached')
    
    # push forward a little bit more just to be certain
    p3dx.sleep(5)
    p3dx.stop()
    
    #lower gripper
    p3dx.gripper(0.05,0.0)
    p3dx.sleep(1)
    
    #open fingers
    p3dx.gripper(0.05,0.1)
    p3dx.sleep(1)
    print('Ball is released')
    #release and back up
    p3dx.move(-1.0,-1.0)
    p3dx.sleep(3)
    p3dx.stop()
    
#turn right by 180 degrees    
def turnRight():
    target = 180.0 * 3.14159265 /180.0     # target angle in radians
    r = 195.3 /2.0      # wheel radius
    L = 330.0           # axis length
    initialEncoder = p3dx.leftEncoder
    robotAngle = 0.0
    while robotAngle < target:
        p3dx.move(1.0,-1.0)
        wheelAngle = p3dx.leftEncoder -initialEncoder 
        robotAngle = 2.0*wheelAngle * r / L
    #p3dx.stop()


def back_up_and_turnaround():
    print('Back up and Turn Around')
    p3dx.move(-2.25,-2.25)
    p3dx.sleep(4) # make sure clear of target area
    turnRight()
